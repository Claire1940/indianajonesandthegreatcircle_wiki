  <Step num="1" title="Speicher überprüfen">

  </Step>
  <Step num="2" title="Starten & Konfigurieren">

  <Step num="3" title="Upscaling aktivieren">

  </Step>
  <Step num="4" title="Speicherverwaltung">
    Verwenden Sie die Steam Cloud, um Ihren Fortschritt zu synchronisieren, insbesondere wenn Sie zwischen dem Steam Deck und einem Desktop-PC wechseln.
  </Step>
